```cpp
#include <bits/stdc++.h>
#define ll long long
using namespace std;

int main() {
	ios::sync_with_stdio(0);
	cin.tie(0);

	ll n;
	cin >> n;

	ll arr[n+10] = {0};
	for (int i = 0; i < n; i++) {
		cin >> arr[i];
	}
	ll Max = 0;
	sort(arr, arr+n, greater<int>());
	for (int i = n-1; i > -1; i--) {
		ll val = arr[i] * (i+1);
		Max = max(Max, val);
	}
	cout << Max;

	return 0;
}
```
