#include <bits/stdc++.h>
#define ll long long
using namespace std;

int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    
    ll n, k, ans=0;
    cin >> n >> k;
    ll A[n+10];
    
    for (ll i=0; i<n; i++) {
        cin >> A[i];
    }
    
    sort(A, A+n);
    for (ll i=0; i<n; i++) {
        ll y = lower_bound(A, A+n, A[i]+k)-A;
        if (y<n) ans+=n-y;
    }
    
    cout << ans;
    
    return 0;
}
