```cpp
#include <bits/stdc++.h>
#define ll long long
using namespace std;

ll A[100010];

int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);

    ll n, ct=0, ans=0, pos=1, diff=0,a=0,start=1, pos2;
    cin >> n;
    for (ll i=1; i<=n; i++) {
        cin >> A[i];
    }

    for (ll i=1; i<=n; i++) {
        a++;
        if (A[i]==1) {
            ct++;
            start = min(start, i);
        } else {
            ct--;
            if (ct==0) {
                if (a>diff) {
                    diff =a;
                    pos2=start;
                }
                a=0;
                start=i+1;
            } 
            
        }
        if (ct>ans) {
            ans=ct;
            pos=i;
        }
    }
    cout << ans <<" "<< pos <<" "<< diff <<" "<< pos2;


    return 0;
}
```
