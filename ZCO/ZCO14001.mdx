```cpp
#include <bits/stdc++.h>
#define ll long long
using namespace std;

ll arr[100010];

int main() {
	ios::sync_with_stdio(0);
	cin.tie(0);

	ll n, h, extra=0, inp, p=0;

	cin >> n >> h;

	for (ll i=0; i<n; i++) {
		cin >> arr[i];
	}

	while (cin >> inp && inp!=0) {
		if (inp==1 && p>0) {
			p--;
		} else if (inp==2 && p<n-1) {
			p++;
		} else if (inp==3 && arr[p]>0 && (!extra) ) {
			extra=1;
			arr[p]--;
		} else if (inp==4 && arr[p]<h && (extra)) {
			extra=0;
			arr[p]++;
		}
	}

	for (ll i=0; i<n; i++) {
		cout << arr[i] << " ";
	}


	return 0;
}
```
