```cpp
#include <bits/stdc++.h>
#define ll long long
using namespace std;

ll F[1010];

int main() {
	ios::sync_with_stdio(0);
	cin.tie(0);

	ll n, ans=0, inp;
	cin >> n;

	for (int i=0; i<n; i++) {
		cin >> inp;
		F[inp]++;
	}

	for (int i=1; i<=1000; i++) {
		for (int j=i+1; j<=1000; j++) {
			ans+=(j-i)*F[i]*F[j];
			//cout << ans << '\n';
		}
	}

	cout << ans;

	return 0;
}
```
